package com.sinosoft.nlis.claim.settlement.enums;

/**
 * 给付状态
 */
public enum BenefitStatus {
    
    JF("0"), // 给付
    JP("1"), // 拒赔
    BCD("2"), // 不承担
    TRPF("4"), // 通融赔付
    XYPF("5"); // 协议赔付

    private String code;

    // Constructor
    BenefitStatus(String code) {
        this.code = code;
    }

    // Getter method
    public String getCode() {
        return code;
    }

}
